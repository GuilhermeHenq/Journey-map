CREATE TABLE journeyPhase (
    journeyPhase_id INT AUTO_INCREMENT PRIMARY KEY,
    journeyMap_id INT,
    linePos INT,
    posX INT,
    length INT,
    description VARCHAR(255),
    emojiTag VARCHAR(255),
    FOREIGN KEY (journeyMap_id) REFERENCES journeyMap(journeyMap_id)
);

CREATE TABLE thought (
    thought_id INT AUTO_INCREMENT PRIMARY KEY,
    journeyMap_id INT,
    linePos INT,
    posX INT,
    length INT,
    description VARCHAR(255),
    emojiTag VARCHAR(255),
    FOREIGN KEY (journeyMap_id) REFERENCES journeyMap(journeyMap_id)
);

CREATE TABLE emotion (
    emotion_id INT AUTO_INCREMENT PRIMARY KEY,
    journeyMap_id INT,
    posX INT,
    lineY INT,
    emojiTag VARCHAR(255),
    FOREIGN KEY (journeyMap_id) REFERENCES journeyMap(journeyMap_id)
);

CREATE TABLE userAction (
    userAction_id INT AUTO_INCREMENT PRIMARY KEY,
    journeyMap_id INT,
    linePos INT,
    posX INT,
    length INT,
    description VARCHAR(255),
    emojiTag VARCHAR(255),
    FOREIGN KEY (journeyMap_id) REFERENCES journeyMap(journeyMap_id)
);

CREATE TABLE contactPoint (
    contactPoint_id INT AUTO_INCREMENT PRIMARY KEY,
    journeyMap_id INT,
    linePos INT,
    posX INT,
    length INT,
    description VARCHAR(255),
    emojiTag VARCHAR(255),
    FOREIGN KEY (journeyMap_id) REFERENCES journeyMap(journeyMap_id)
);
